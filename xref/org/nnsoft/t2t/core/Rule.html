16  package org.nnsoft.t2t.core;
17  
18  import org.openrdf.model.Statement;
19  import org.openrdf.model.Value;
20  import org.openrdf.model.impl.StatementImpl;
21  import org.openrdf.model.impl.URIImpl;
22  import org.openrdf.query.algebra.StatementPattern;
23  import org.openrdf.query.algebra.Var;
24  
25  import java.util.HashSet;
26  import java.util.Set;
27  
28  /**
29   * @author Davide Palmisano ( dpalmisano@gmail.com )
30   */
31  public class Rule {
32  
33      private StatementPattern match;
34  
35      private Set<StatementPattern> apply;
36  
37      public Rule(StatementPattern match, Set<StatementPattern> apply) {
38          this.match = match;
39          this.apply = apply;
40      }
41  
42      public boolean match(Statement statement)
43              throws RuleExecutionException {
44          Var sVar = match.getSubjectVar();
45          Var pVar = match.getPredicateVar();
46          Var oVar = match.getObjectVar();
47          if(sVar.getValue() != null && !sVar.getValue().equals(statement.getSubject())) {
48              return false;
49          }
50          if(pVar.getValue() != null && !pVar.getValue().equals(statement.getPredicate())) {
51              return false;
52          }
53          if(oVar.getValue() != null && !oVar.getValue().equals(statement.getObject())) {
54              return false;
55          }
56          return true;
57      }
58  
59      public Set<Statement> apply(Statement statement) 
60              throws RuleExecutionException {
61          Set<Statement> result = new HashSet<Statement>();
62          if(match(statement)) {
63              // bind the statement values to the match pattern
64              Var[] bindings = getBindings(match, statement);
65              // System.out.println("applaying rule!");
66              for(StatementPattern apply : this.apply) {
67                  Var applySVar = apply.getSubjectVar();
68                  Var applyPVar = apply.getPredicateVar();
69                  Var applyOVar = apply.getObjectVar();
70                  Value subjectValue = getBindValue(bindings, applySVar);
71                  Value predicateValue = getBindValue(bindings, applyPVar);
72                  Value objectValue = getBindValue(bindings, applyOVar);
73                  Statement newStatement = new StatementImpl(
74                          new URIImpl(subjectValue.stringValue()),
75                          new URIImpl(predicateValue.stringValue()),
76                          objectValue
77                  );
78                  result.add(newStatement);
79              }
80              return result;
81          }
82          return result;
83      }
84  
85      private Var[] getBindings(StatementPattern match, Statement statement) {
86          Var sVar = new Var(match.getSubjectVar().getName(), statement.getSubject());
87          Var pVar = new Var(match.getPredicateVar().getName(), statement.getPredicate());
88          Var oVar = new Var(match.getObjectVar().getName(), statement.getObject());
89          Var[] bindings = new Var[3];
90          bindings[0] = sVar;
91          bindings[1] = pVar;
92          bindings[2] = oVar;
93          return bindings;
94      }
95  
96      private Value getBindValue(Var[] bindings, Var var) throws RuleExecutionException {
97          if(var.isAnonymous() && var.getValue() != null) {
98              return var.getValue();
99          }
100         for(Var bind : bindings) {
101             if(bind.getName().equals(var.getName()))
102                 return bind.getValue();
103         }
104         throw new RuleExecutionException("");
105     }
106 
107     @Override
108     public String toString() {
109         return "Rule{" +
110                 "match=" + match +
111                 ", apply=" + apply +
112                 '}';
113     }
114 }
